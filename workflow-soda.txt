import json
import time
import threading
from datetime import datetime, timedelta
from enum import Enum
from typing import Dict, List, Callable, Any, Optional
from dataclasses import dataclass, field


class WorkflowStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    PAUSED = "paused"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"


class StepStatus(Enum):
    WAITING = "waiting"
    RUNNING = "running"
    COMPLETED = "completed"
    SKIPPED = "skipped"
    FAILED = "failed"


@dataclass
class WorkflowStep:
    """Single step in a workflow"""
    name: str
    description: str  # What SODA will explain to user
    action: str  # Function/action to execute
    params: Dict = field(default_factory=dict)
    condition: Optional[str] = None  # Condition to check before running
    wait_time: float = 0  # Seconds to wait before next step
    show_visual: bool = True  # Show visual window for this step
    auto_close_visual: bool = True  # Auto close the visual after explanation
    visual_duration: float = 5.0  # How long to show visual
    status: StepStatus = StepStatus.WAITING
    result: Any = None
    error: str = ""
    
    def to_dict(self):
        return {
            "name": self.name,
            "description": self.description,
            "action": self.action,
            "params": self.params,
            "status": self.status.value,
            "result": str(self.result) if self.result else None
        }


@dataclass
class Workflow:
    """Complete workflow with multiple steps"""
    id: str
    name: str
    trigger: str  # What triggers this workflow
    description: str
    steps: List[WorkflowStep] = field(default_factory=list)
    status: WorkflowStatus = WorkflowStatus.PENDING
    context: Dict = field(default_factory=dict)  # Shared data between steps
    created_at: datetime = field(default_factory=datetime.now)
    completed_at: Optional[datetime] = None
    current_step_index: int = 0
    
    def add_step(self, step: WorkflowStep):
        self.steps.append(step)
        return self
    
    def get_current_step(self) -> Optional[WorkflowStep]:
        if self.current_step_index < len(self.steps):
            return self.steps[self.current_step_index]
        return None
    
    def get_progress(self) -> float:
        if not self.steps:
            return 0
        completed = sum(1 for s in self.steps if s.status in [StepStatus.COMPLETED, StepStatus.SKIPPED])
        return (completed / len(self.steps)) * 100
    
    def to_dict(self):
        return {
            "id": self.id,
            "name": self.name,
            "trigger": self.trigger,
            "status": self.status.value,
            "progress": self.get_progress(),
            "current_step": self.current_step_index,
            "total_steps": len(self.steps),
            "steps": [s.to_dict() for s in self.steps]
        }


class WorkflowEngine:
    """Main engine that manages and executes workflows"""
    
    def __init__(self, soda_core):
        self.soda = soda_core  # Reference to main SODA instance
        self.workflows: Dict[str, Workflow] = {}
        self.registered_workflows: Dict[str, Callable] = {}  # Workflow templates
        self.action_registry: Dict[str, Callable] = {}  # Available actions
        self.running_workflow: Optional[Workflow] = None
        self.workflow_history: List[Dict] = []
        self.is_executing = False
        self._lock = threading.Lock()
        
        # Register all built-in actions
        self._register_built_in_actions()
    
    def _register_built_in_actions(self):

    def register_action(self, name: str, func: Callable):
        """Register an action that can be used in workflows"""
        self.action_registry[name] = func
    
    def register_workflow_template(self, trigger: str, builder: Callable):
        """Register a workflow template that can be triggered"""
        self.registered_workflows[trigger] = builder
    
    def find_matching_workflow(self, user_input: str) -> Optional[str]:
        """Use AI to determine which workflow matches user input"""
        # This will be called by the main SODA to detect workflow triggers
        triggers = list(self.registered_workflows.keys())
        
        # Simple keyword matching first
        user_lower = user_input.lower()
        
        trigger_keywords = {
            "morning_briefing": ["good morning", "morning", "wake up", "start my day", "briefing"],
            "going_outside": ["going outside", "going out", "leaving", "heading out", "stepping out", "i'm going"],
            "project_review": ["project review", "check projects", "github", "show my projects", "project status"],
            "work_session": ["start working", "work mode", "let's work", "focus mode", "start coding"],
            "end_of_day": ["good night", "end of day", "shutting down", "closing time", "i'm done"],
            "health_check": ["system health", "check system", "how's my pc", "system status", "diagnostics"],
            "meeting_prep": ["meeting in", "prepare for meeting", "meeting prep", "joining meeting"],
            "code_review": ["review code", "code review", "check my code", "analyze code"],
            "deploy_check": ["deploy", "deployment", "push to production", "release"],
            "break_time": ["taking a break", "break time", "need a break", "pause"],
            "startup": ["__startup__"],  # Internal trigger
        }
        
        for workflow_name, keywords in trigger_keywords.items():
            for keyword in keywords:
                if keyword in user_lower:
                    if workflow_name in self.registered_workflows:
                        return workflow_name
        
        return None
    
    def create_workflow(self, trigger: str, context: Dict = None) -> Optional[Workflow]:
        """Create a workflow instance from a template"""
        if trigger not in self.registered_workflows:
            return None
        
        builder = self.registered_workflows[trigger]
        workflow = builder(context or {})
        self.workflows[workflow.id] = workflow
        return workflow
    
    async def execute_workflow(self, workflow: Workflow, callback=None):
        """Execute a workflow step by step"""
        with self._lock:
            if self.is_executing:
                self.soda.speak("I'm already running a workflow. Please wait for it to finish.")
                return
            self.is_executing = True
        
        self.running_workflow = workflow
        workflow.status = WorkflowStatus.RUNNING
        
        try:
            # Announce workflow start
            await self._action_speak({
                "text": f"Starting workflow: {workflow.name}. {workflow.description}. "
                       f"This has {len(workflow.steps)} steps. Let me walk you through everything."
            }, workflow)
            
            if callback:
                callback("workflow_started", workflow.to_dict())
            
            # Execute each step
            for i, step in enumerate(workflow.steps):
                workflow.current_step_index = i
                
                if workflow.status == WorkflowStatus.CANCELLED:
                    break
                
                # Check condition
                if step.condition:
                    should_run = self._evaluate_condition(step.condition, workflow.context)
                    if not should_run:
                        step.status = StepStatus.SKIPPED
                        await self._action_speak({
                            "text": f"Skipping step: {step.name}. Condition not met."
                        }, workflow)
                        continue
                
                # Announce step
                step.status = StepStatus.RUNNING
                await self._action_speak({
                    "text": f"Step {i+1} of {len(workflow.steps)}: {step.description}"
                }, workflow)
                
                if callback:
                    callback("step_started", {"step": step.to_dict(), "index": i})
                
                # Show visual if needed
                if step.show_visual:
                    await self._action_show_visual({
                        "title": step.name,
                        "description": step.description,
                        "step": i + 1,
                        "total": len(workflow.steps),
                        "progress": workflow.get_progress()
                    }, workflow)
                
                # Execute the action
                try:
                    if step.action in self.action_registry:
                        action_func = self.action_registry[step.action]
                        result = await action_func(step.params, workflow)
                        step.result = result
                        step.status = StepStatus.COMPLETED
                        
                        # Store result in workflow context
                        workflow.context[f"step_{i}_result"] = result
                        workflow.context[step.name] = result
                        
                        # Explain the result
                        if result and isinstance(result, dict) and "explanation" in result:
                            await self._action_speak({
                                "text": result["explanation"]
                            }, workflow)
                    else:
                        step.status = StepStatus.FAILED
                        step.error = f"Action '{step.action}' not found"
                        
                except Exception as e:
                    step.status = StepStatus.FAILED
                    step.error = str(e)
                    await self._action_speak({
                        "text": f"Step {step.name} encountered an issue: {str(e)}. Moving to next step."
                    }, workflow)
                
                # Auto close visual
                if step.show_visual and step.auto_close_visual:
                    await self._action_wait({"seconds": step.visual_duration}, workflow)
                    await self._action_close_visual({}, workflow)
                
                # Wait between steps
                if step.wait_time > 0:
                    await self._action_wait({"seconds": step.wait_time}, workflow)
                
                if callback:
                    callback("step_completed", {"step": step.to_dict(), "index": i})
            
            # Workflow complete
            workflow.status = WorkflowStatus.COMPLETED
            workflow.completed_at = datetime.now()
            
            # Final summary
            completed_steps = sum(1 for s in workflow.steps if s.status == StepStatus.COMPLETED)
            skipped_steps = sum(1 for s in workflow.steps if s.status == StepStatus.SKIPPED)
            failed_steps = sum(1 for s in workflow.steps if s.status == StepStatus.FAILED)
            
            await self._action_speak({
                "text": f"Workflow '{workflow.name}' completed! "
                       f"{completed_steps} steps completed, {skipped_steps} skipped, {failed_steps} failed. "
                       f"Total time: {(workflow.completed_at - workflow.created_at).seconds} seconds."
            }, workflow)
            
            if callback:
                callback("workflow_completed", workflow.to_dict())
                
        except Exception as e:
            workflow.status = WorkflowStatus.FAILED
            await self._action_speak({
                "text": f"Workflow failed with error: {str(e)}"
            }, workflow)
            
        finally:
            self.running_workflow = None
            self.is_executing = False
            self.workflow_history.append(workflow.to_dict())
    
    def cancel_current_workflow(self):
        """Cancel the currently running workflow"""
        if self.running_workflow:
            self.running_workflow.status = WorkflowStatus.CANCELLED
    
    def _evaluate_condition(self, condition: str, context: Dict) -> bool:
        """Evaluate a condition string against workflow context"""
        try:
            # Simple condition evaluation
            # Examples: "weather.is_rainy == True", "battery_level < 20"
            return eval(condition, {"__builtins__": {}}, context)
        except:
            return True  # Default to running the step

    async def _action_check_weather(self, params: Dict, workflow: Workflow) -> Dict:
        """Check current weather"""
        import aiohttp
        
        location = params.get("location", workflow.context.get("location", "auto"))
        
        try:
            # Using wttr.in for simple weather (no API key needed)
            async with aiohttp.ClientSession() as session:
                async with session.get(f"https://wttr.in/{location}?format=j1") as resp:
                    if resp.status == 200:
                        data = await resp.json()
                        current = data["current_condition"][0]
                        
                        weather_info = {
                            "temperature": current["temp_C"],
                            "feels_like": current["FeelsLikeC"],
                            "humidity": current["humidity"],
                            "description": current["weatherDesc"][0]["value"],
                            "wind_speed": current["windspeedKmph"],
                            "uv_index": current.get("uvIndex", "N/A"),
                            "is_rainy": "rain" in current["weatherDesc"][0]["value"].lower(),
                            "is_sunny": "sun" in current["weatherDesc"][0]["value"].lower() or 
                                       "clear" in current["weatherDesc"][0]["value"].lower(),
                            "is_cloudy": "cloud" in current["weatherDesc"][0]["value"].lower(),
                        }
                        
                        # Smart suggestions
                        suggestions = []
                        if weather_info["is_rainy"]:
                            suggestions.append("Take an umbrella")
                        if int(weather_info.get("uv_index", 0)) > 5:
                            suggestions.append("Apply sunscreen - UV index is high")
                        if int(weather_info["temperature"]) < 10:
                            suggestions.append("Wear a warm jacket")
                        if int(weather_info["temperature"]) > 35:
                            suggestions.append("Stay hydrated, carry water")
                        if int(weather_info["wind_speed"]) > 30:
                            suggestions.append("It's quite windy outside")
                        
                        weather_info["suggestions"] = suggestions
                        weather_info["explanation"] = (
                            f"Current weather: {weather_info['description']}, "
                            f"{weather_info['temperature']}°C, feels like {weather_info['feels_like']}°C. "
                            f"Humidity is {weather_info['humidity']}%. "
                            f"{'My suggestions: ' + '. '.join(suggestions) if suggestions else 'Looks good to go!'}"
                        )
                        
                        # Store in workflow context
                        workflow.context["weather"] = weather_info
                        return weather_info
                        
        except Exception as e:
            return {
                "error": str(e),
                "explanation": "I couldn't fetch the weather data right now. You might want to check manually."
            }
    
    async def _action_check_github_repos(self, params: Dict, workflow: Workflow) -> Dict:
        """Scan GitHub repositories"""
        import aiohttp
        
        username = params.get("username", "your-github-username")
        token = params.get("token", "")
        
        try:
            headers = {"Authorization": f"token {token}"} if token else {}
            
            async with aiohttp.ClientSession() as session:
                # Get all repos
                async with session.get(
                    f"https://api.github.com/users/{username}/repos?sort=updated&per_page=100",
                    headers=headers
                ) as resp:
                    repos = await resp.json()
                
                repo_stats = []
                total_stars = 0
                total_forks = 0
                
                for repo in repos:
                    if isinstance(repo, dict):
                        stars = repo.get("stargazers_count", 0)
                        forks = repo.get("forks_count", 0)
                        total_stars += stars
                        total_forks += forks
                        
                        repo_stats.append({
                            "name": repo["name"],
                            "stars": stars,
                            "forks": forks,
                            "language": repo.get("language", "Unknown"),
                            "description": repo.get("description", ""),
                            "updated_at": repo.get("updated_at", ""),
                            "open_issues": repo.get("open_issues_count", 0),
                            "url": repo.get("html_url", ""),
                            "is_fork": repo.get("fork", False),
                            "size": repo.get("size", 0),
                        })
                
                # Sort by stars (ranking)
                repo_stats.sort(key=lambda x: x["stars"], reverse=True)
                
                # Top repos
                top_repos = repo_stats[:5]
                
                result = {
                    "total_repos": len(repo_stats),
                    "total_stars": total_stars,
                    "total_forks": total_forks,
                    "top_repos": top_repos,
                    "all_repos": repo_stats,
                    "most_used_language": max(
                        set(r["language"] for r in repo_stats if r["language"]),
                        key=lambda l: sum(1 for r in repo_stats if r["language"] == l)
                    ) if repo_stats else "None",
                    "explanation": (
                        f"You have {len(repo_stats)} repositories on GitHub. "
                        f"Total stars: {total_stars}, Total forks: {total_forks}. "
                        f"Your top project is '{top_repos[0]['name']}' with {top_repos[0]['stars']} stars. "
                        f"Most used language: {max(set(r['language'] for r in repo_stats if r['language']), key=lambda l: sum(1 for r in repo_stats if r['language'] == l)) if repo_stats else 'None'}."
                    )
                }
                
                workflow.context["github_repos"] = result
                return result
                
        except Exception as e:
            return {
                "error": str(e),
                "explanation": f"Couldn't fetch GitHub data: {str(e)}"
            }

    async def _action_scan_project_progress(self, params: Dict, workflow: Workflow) -> Dict:
        """Scan a project directory to estimate progress"""
        import os
        import re
        
        project_path = params.get("path", ".")
        project_name = params.get("name", os.path.basename(project_path))
        
        try:
            stats = {
                "total_files": 0,
                "total_lines": 0,
                "todo_count": 0,
                "fixme_count": 0,
                "hack_count": 0,
                "completed_features": [],
                "pending_todos": [],
                "file_types": {},
                "largest_files": [],
            }
            
            ignore_dirs = {'.git', 'node_modules', '__pycache__', 'venv', '.venv', 
                          'build', 'dist', '.next', '.cache'}
            code_extensions = {'.py', '.js', '.ts', '.jsx', '.tsx', '.java', '.cpp', 
                             '.c', '.go', '.rs', '.dart', '.swift', '.kt'}
            
            for root, dirs, files in os.walk(project_path):
                # Skip ignored directories
                dirs[:] = [d for d in dirs if d not in ignore_dirs]
                
                for file in files:
                    filepath = os.path.join(root, file)
                    ext = os.path.splitext(file)[1]
                    
                    stats["total_files"] += 1
                    stats["file_types"][ext] = stats["file_types"].get(ext, 0) + 1
                    
                    if ext in code_extensions:
                        try:
                            with open(filepath, 'r', encoding='utf-8', errors='ignore') as f:
                                lines = f.readlines()
                                stats["total_lines"] += len(lines)
                                
                                file_size = len(lines)
                                stats["largest_files"].append({
                                    "file": os.path.relpath(filepath, project_path),
                                    "lines": file_size
                                })
                                
                                for line_num, line in enumerate(lines, 1):
                                    line_upper = line.upper()
                                    if "TODO" in line_upper:
                                        stats["todo_count"] += 1
                                        stats["pending_todos"].append({
                                            "file": os.path.relpath(filepath, project_path),
                                            "line": line_num,
                                            "text": line.strip()
                                        })
                                    if "FIXME" in line_upper:
                                        stats["fixme_count"] += 1
                                    if "HACK" in line_upper:
                                        stats["hack_count"] += 1
                        except:
                            pass
            
            # Sort largest files
            stats["largest_files"].sort(key=lambda x: x["lines"], reverse=True)
            stats["largest_files"] = stats["largest_files"][:10]
            
            # Estimate completion (rough heuristic)
            total_issues = stats["todo_count"] + stats["fixme_count"]
            if total_issues == 0:
                estimated_completion = 95  # Almost done!
            elif total_issues < 5:
                estimated_completion = 85
            elif total_issues < 15:
                estimated_completion = 70
            elif total_issues < 30:
                estimated_completion = 50
            else:
                estimated_completion = 30
            
            stats["estimated_completion"] = estimated_completion
            stats["project_name"] = project_name
            
            stats["explanation"] = (
                f"Project '{project_name}' analysis: "
                f"{stats['total_files']} files, {stats['total_lines']} lines of code. "
                f"Found {stats['todo_count']} TODOs and {stats['fixme_count']} FIXMEs. "
                f"Estimated completion: {estimated_completion}%. "
                f"{'The project looks nearly complete!' if estimated_completion > 80 else ''}"
                f"{'Quite a bit of work remaining.' if estimated_completion < 50 else ''}"
                f" Top pending items: {', '.join(t['text'][:50] for t in stats['pending_todos'][:3]) if stats['pending_todos'] else 'None found.'}"
            )
            
            workflow.context["project_progress"] = stats
            return stats
            
        except Exception as e:
            return {"error": str(e), "explanation": f"Couldn't scan project: {str(e)}"}

        memory = psutil.virtual_memory()
        disk = psutil.disk_usage('/')
        battery = psutil.sensors_battery()
        
        # Network
        net_io = psutil.net_io_counters()
        
        # Running processes
        top_processes = []
        for proc in sorted(psutil.process_iter(['name', 'cpu_percent', 'memory_percent']),
                          key=lambda p: p.info.get('cpu_percent', 0) or 0, reverse=True)[:5]:
            top_processes.append({
                "name": proc.info['name'],
                "cpu": proc.info.get('cpu_percent', 0),
                "memory": round(proc.info.get('memory_percent', 0), 1)
            })
        
        system_info = {
            "cpu_percent": cpu_percent,
            "memory_total_gb": round(memory.total / (1024**3), 1),
            "memory_used_gb": round(memory.used / (1024**3), 1),
            "memory_percent": memory.percent,
            "disk_total_gb": round(disk.total / (1024**3), 1),
            "disk_used_gb": round(disk.used / (1024**3), 1),
            "disk_percent": disk.percent,
            "battery_percent": battery.percent if battery else "N/A",
            "battery_plugged": battery.power_plugged if battery else "N/A",
            "top_processes": top_processes,
            "network_sent_mb": round(net_io.bytes_sent / (1024**2), 1),
            "network_recv_mb": round(net_io.bytes_recv / (1024**2), 1),
        }
        
        # Health assessment
        issues = []
        if cpu_percent > 80:
            issues.append(f"CPU usage is high at {cpu_percent}%")
        if memory.percent > 85:
            issues.append(f"Memory usage is high at {memory.percent}%")
        if disk.percent > 90:
            issues.append(f"Disk space is running low at {disk.percent}%")
        if battery and battery.percent < 20 and not battery.power_plugged:
            issues.append(f"Battery is low at {battery.percent}%")
        
        system_info["health_status"] = "Good" if not issues else "Needs Attention"
        system_info["issues"] = issues
        
        system_info["explanation"] = (
            f"System Status: CPU at {cpu_percent}%, "
            f"Memory: {memory.percent}% used ({round(memory.used/(1024**3),1)}GB of {round(memory.total/(1024**3),1)}GB), "
            f"Disk: {disk.percent}% used. "
            f"{'Battery: ' + str(battery.percent) + '%, ' + ('charging' if battery.power_plugged else 'on battery') + '. ' if battery else ''}"
            f"{'Issues found: ' + '. '.join(issues) if issues else 'Everything looks healthy!'}"
        )
        
        workflow.context["system_status"] = system_info
        return system_info

import tkinter as tk
from tkinter import ttk
import threading

import time


class WorkflowVisualWindow:
    """Animated visual windows for workflow steps"""
    
    def __init__(self):
        self.current_window = None
        self.is_showing = False
        self._animation_running = False
    
    def show_step_visual(self, step_data: dict):
        """Show an animated visual for a workflow step"""
        if self.current_window:
            self.close_visual()
        
        self._animation_running = True
        thread = threading.Thread(target=self._create_visual, args=(step_data,), daemon=True)
        thread.start()
    
    def close_visual(self):
        """Close current visual with animation"""
        self._animation_running = False
        if self.current_window:
            try:
                self.current_window.destroy()
            except:
                pass
            self.current_window = None
            self.is_showing = False
    
    def _create_visual(self, data: dict):
        """Create the visual window"""
        root = tk.Tk()
        self.current_window = root
        self.is_showing = True
        
        # Window setup
        width = 500
        height = 400
        screen_w = root.winfo_screenwidth()
        screen_h = root.winfo_screenheight()
        x = screen_w - width - 20
        y = screen_h - height - 80
        
        root.geometry(f"{width}x{height}+{x}+{y}")
        root.overrideredirect(True)  # No title bar
        root.attributes('-topmost', True)
        root.attributes('-alpha', 0.0)  # Start transparent
        
        # Dark theme
        bg_color = "#1a1a2e"
        accent_color = "#e94560"
        text_color = "#eee"
        secondary_text = "#aaa"
        
        root.configure(bg=bg_color)
        
        # Main frame
        main_frame = tk.Frame(root, bg=bg_color, padx=20, pady=15)
        main_frame.pack(fill=tk.BOTH, expand=True)
        
        # Step indicator
        step_num = data.get("step", 1)
        total_steps = data.get("total", 1)
        progress = data.get("progress", 0)
        
        # Header
        header_frame = tk.Frame(main_frame, bg=bg_color)
        header_frame.pack(fill=tk.X, pady=(0, 10))

            fg=accent_color,
            bg=bg_color

        close_btn = tk.Label(
            header_frame,
            text="✕",
            font=("Segoe UI", 12),
            fg="#666",
            bg=bg_color,

        close_btn.bind("<Button-1>", lambda e: self.close_visual())
        
        # Title
        title_label = tk.Label(
            main_frame,

            font=("Segoe UI", 16, "bold"),
            fg=text_color,
            bg=bg_color,

        )
        title_label.pack(fill=tk.X, pady=(0, 5))
        
        # Description
        desc_label = tk.Label(

            text=data.get("description", ""),
            font=("Segoe UI", 10),
            fg=secondary_text,
            bg=bg_color,
            wraplength=450,
            justify=tk.LEFT
        )
        desc_label.pack(fill=tk.X, pady=(0, 15))
        
        # Progress bar
        progress_frame = tk.Frame(main_frame, bg=bg_color)
        progress_frame.pack(fill=tk.X, pady=(0, 10))
        
        # Custom progress bar using canvas
        progress_canvas = tk.Canvas(
            progress_frame,
            height=6,
            bg="#333",
            highlightthickness=0
        )
        progress_canvas.pack(fill=tk.X)
        
        # Content area (for displaying data)
        content_frame = tk.Frame(main_frame, bg="#16213e", padx=15, pady=15)
        content_frame.pack(fill=tk.BOTH, expand=True, pady=(0, 10))
        
        # Display result data if available
        result = data.get("result", {})
        if isinstance(result, dict):
            explanation = result.get("explanation", "Processing...")
            content_label = tk.Label(
                content_frame,
                text=explanation,
                font=("Segoe UI", 11),
                fg=text_color,
                bg="#16213e",
                wraplength=430,
                justify=tk.LEFT
            )
            content_label.pack(fill=tk.BOTH, expand=True)
        else:
            loading_label = tk.Label(
                content_frame,
                text="⟳ Processing...",
                font=("Segoe UI", 14),
                fg=accent_color,
                bg="#16213e"
            )
            loading_label.pack(expand=True)
        
        # SODA branding
        brand_label = tk.Label(
            main_frame,
            text="🤖 SODA AI Workflow Engine",
            font=("Segoe UI", 8),
            fg="#555",
            bg=bg_color
        )
        brand_label.pack(side=tk.BOTTOM)
        
        # Fade in animation
        def fade_in(alpha=0.0):
            if alpha < 0.95 and self._animation_running:
                alpha += 0.05
                root.attributes('-alpha', alpha)
                root.after(20, lambda: fade_in(alpha))
            else:
                root.attributes('-alpha', 0.95)
                # Draw progress bar after fade in
                draw_progress(0)
        
        def draw_progress(current):
            if current <= progress and self._animation_running:
                progress_canvas.delete("all")
                canvas_width = progress_canvas.winfo_width()
                if canvas_width > 1:

                        0, 0, bar_width, 6,
                        fill=accent_color,
                        outline=""
                    )
                root.after(30, lambda: draw_progress(current + 1))

        root.after(50, fade_in)
        
        # Make window draggable
        def start_drag(event):
            root._drag_x = event.x
            root._drag_y = event.y
        
        def drag(event):
            x = root.winfo_x() + event.x - root._drag_x
            y = root.winfo_y() + event.y - root._drag_y
            root.geometry(f"+{x}+{y}")
        
        header_frame.bind("<Button-1>", start_drag)
        header_frame.bind("<B1-Motion>", drag)
        
        try:
            root.mainloop()
        except:
            pass

# soda_with_workflows.py

import asyncio
from workflow_engine import WorkflowEngine
from workflow_templates import register_all_workflows
from workflow_visuals import WorkflowVisualWindow, WorkflowProgressOverlay


class SODA:
    """Main SODA class with workflow integration"""
    
    def __init__(self):
        # Your existing SODA initialization...
        self.name = "SODA"
        
        # Workflow Engine
        self.workflow_engine = WorkflowEngine(self)
        self.visual_window = WorkflowVisualWindow()
        self.progress_overlay = WorkflowProgressOverlay()
        
        # Configuration
        self.config = {
            "github_username": "your-github-username",
            "github_token": "",  # Optional for private repos
            "project_path": r"C:\Users\you\Projects\soda-ai",
            "project_name": "SODA AI",
            "ide": "code",  # VS Code
        }
        
        # Register all workflow templates
        register_all_workflows(self.workflow_engine)
        
        # Override the visual actions to use our visual system
        self.workflow_engine.register_action("show_visual", self._show_workflow_visual)
        self.workflow_engine.register_action("close_visual", self._close_workflow_visual)
    
    async def process_input(self, user_input: str):
        """Process user input - check for workflows first, then single commands"""
        
        # Check if input matches any workflow
        workflow_trigger = self.workflow_engine.find_matching_workflow(user_input)
        
        if workflow_trigger:
            # Create and execute workflow
            workflow = self.workflow_engine.create_workflow(
                workflow_trigger,
                context=self.config
            )
            
            if workflow:
                # Show progress overlay
                self.progress_overlay.show(workflow.to_dict())
                
                # Execute workflow
                await self.workflow_engine.execute_workflow(
                    workflow,
                    callback=self._workflow_callback
                )
                
                # Close progress overlay
                self.progress_overlay.close()
                return
        
        # If no workflow matched, handle as single command (your existing logic)
        await self._handle_single_command(user_input)
    
    async def on_startup(self):
        """Called when SODA starts up"""
        # Run startup workflow
        workflow = self.workflow_engine.create_workflow("startup", context=self.config)
        if workflow:
            await self.workflow_engine.execute_workflow(
                workflow,
                callback=self._workflow_callback
            )
    
    def _workflow_callback(self, event: str, data: dict):
        """Callback for workflow events"""
        if event == "step_started":
            # Update visual
            self.visual_window.show_step_visual(data.get("step", {}))
            # Update progress overlay
            self.progress_overlay.update(data)
        elif event == "step_completed":
            pass
        elif event == "workflow_completed":
            print(f"✅ Workflow completed: {data.get('name')}")
    
    async def _show_workflow_visual(self, params: dict, workflow) -> dict:

# custom_workflow_builder.py

import json
import uuid
from workflow_engine import Workflow, WorkflowStep


class WorkflowBuilder:
    """Allows creating custom workflows through conversation or config"""
    
    def __init__(self, engine):
        self.engine = engine
    
    def create_from_config(self, config: dict) -> Workflow:
        """Create workflow from a JSON config"""
        workflow = Workflow(
            id=f"custom_{uuid.uuid4().hex[:8]}",
            name=config.get("name", "Custom Workflow"),
            trigger=config.get("trigger", "custom"),
            description=config.get("description", "A custom workflow"),
        )
        
        for step_config in config.get("steps", []):
            step = WorkflowStep(
                name=step_config["name"],
                description=step_config.get("description", ""),
                action=step_config["action"],
                params=step_config.get("params", {}),
                condition=step_config.get("condition"),
                wait_time=step_config.get("wait_time", 0),
                show_visual=step_config.get("show_visual", True),
                auto_close_visual=step_config.get("auto_close_visual", True),
                visual_duration=step_config.get("visual_duration", 5),
            )
            workflow.add_step(step)
        
        return workflow
    
    def save_workflow(self, workflow_config: dict, filename: str = "custom_workflows.json"):
        """Save a custom workflow to file"""
        try:
            with open(filename, 'r') as f:
                workflows = json.load(f)
        except FileNotFoundError:
            workflows = []
        
        workflows.append(workflow_config)
        
        with open(filename, 'w') as f:
            json.dump(workflows, f, indent=2)
    
    def load_custom_workflows(self, filename: str = "custom_workflows.json"):
        """Load and register all custom workflows"""
        try:
            with open(filename, 'r') as f:
                workflows = json.load(f)
            
            for wf_config in workflows:
                trigger = wf_config.get("trigger", "")
                if trigger:
                    self.engine.register_workflow_template(
                        trigger,
                        lambda ctx, cfg=wf_config: self.create_from_config(cfg)
                    )
                    
        except FileNotFoundError:
            pass
