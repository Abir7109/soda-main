Step 3 — Write the API (main.py)
pythonfrom fastapi import FastAPI, UploadFile, File, Form
from fastapi.responses import JSONResponse
import face_recognition
import numpy as np
import json, os, io
from PIL import Image

app = FastAPI()
MEMORY_FILE = "face_memory.json"

def load_memory():
    if not os.path.exists(MEMORY_FILE):
        return []
    with open(MEMORY_FILE, "r") as f:
        return json.load(f)

def save_memory(data):
    with open(MEMORY_FILE, "w") as f:
        json.dump(data, f)

def image_to_encoding(file_bytes):
    image = face_recognition.load_image_file(io.BytesIO(file_bytes))
    encodings = face_recognition.face_encodings(image)
    return encodings[0] if encodings else None


@app.post("/recognize")
async def recognize(file: UploadFile = File(...)):
    contents = await file.read()
    encoding = image_to_encoding(contents)
    if encoding is None:
        return JSONResponse({"name": None, "message": "No face detected"})
    
    memory = load_memory()
    for person in memory:
        known = np.array(person["embedding"])
        match = face_recognition.compare_faces([known], encoding, tolerance=0.6)
        if match[0]:
            return JSONResponse({"name": person["name"]})
    
    return JSONResponse({"name": None, "message": "Face not recognized"})


@app.post("/remember")
async def remember(file: UploadFile = File(...), name: str = Form(...)):
    contents = await file.read()
    encoding = image_to_encoding(contents)
    if encoding is None:
        return JSONResponse({"success": False, "message": "No face detected"})
    
    memory = load_memory()
    memory.append({"name": name, "embedding": encoding.tolist()})
    save_memory(memory)
    return JSONResponse({"success": True, "message": f"Remembered {name}"})

Step 9 — Add to Soda's external_apis.py
pythonimport httpx
import base64
import cv2
import os

FACE_API_URL = os.getenv("FACE_API_URL", "https://soda-face-api.up.railway.app")

def capture_frame_bytes():
    cam = cv2.VideoCapture(0)
    ret, frame = cam.read()
    cam.release()
    if not ret:
        return None
    _, buffer = cv2.imencode('.jpg', frame)
    return buffer.tobytes()

async def recognize_face() -> dict:
    image_bytes = capture_frame_bytes()
    if not image_bytes:
        return {"name": None, "message": "Could not access webcam"}
    async with httpx.AsyncClient(timeout=15) as client:
        resp = await client.post(
            f"{FACE_API_URL}/recognize",
            files={"file": ("frame.jpg", image_bytes, "image/jpeg")}
        )
        return resp.json()

async def remember_face(name: str) -> dict:
    image_bytes = capture_frame_bytes()
    if not image_bytes:
        return {"success": False, "message": "Could not access webcam"}
    async with httpx.AsyncClient(timeout=15) as client:
        resp = await client.post(
            f"{FACE_API_URL}/remember",
            files={"file": ("frame.jpg", image_bytes, "image/jpeg")},
            data={"name": name}
        )
        return resp.json()

pythonelif fc.name == 'recognize_face':
    result = await external_apis.recognize_face()
    if result.get("name"):
        tool_result = f"I can see {result['name']}."
    else:
        tool_result = "I don't recognize this person. Tell me their name and I'll remember them."

elif fc.name == 'remember_face':
    name = fc.args.get('name', '')
    result = await external_apis.remember_face(name)
    tool_result = f"Got it, I'll remember them as {name}." if result.get("success") else "I couldn't detect a face clearly."

That's the complete pipeline. Soda captures one webcam frame → sends it to Railway → Railway identifies the person → Soda speaks the name. Your machine does nothing heavy. Railway handles all of it.
