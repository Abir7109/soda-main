class WebBuilderAgent:
    """
    Orchestrates the entire website building flow:
    1. Interview the user
    2. Generate optimized prompt
    3. Automate the builder
    4. Monitor and report
    """

    def __init__(self, sio, gemini_client=None):
        self.sio = sio
        self.gemini_client = gemini_client
        self.session_data = {}
        self.current_question_index = 0
        self.question_queue = []
        self.interview_phase = "not_started"
        self.selected_builder = None
        self.generated_prompt = None
        self.build_status = "idle"

    def start_interview(self) -> str:
        """Begin the interview process. Returns the first question."""
        self.session_data = {}
        self.interview_phase = "universal"
        self.question_queue = list(QUESTION_TREE["universal"])
        self.current_question_index = 0

        first_q = self.question_queue[0]
        return first_q["question"]

    def process_answer(self, answer: str) -> dict:
        """
        Process user's answer and return next question or completion status.
        
        Returns:
            {
                "status": "next_question" | "interview_complete",
                "question": str | None,
                "question_id": str,
                "progress": float  # 0.0 to 1.0
            }
        """
        if self.current_question_index < len(self.question_queue):
            current_q = self.question_queue[self.current_question_index]
            self.session_data[current_q["extract_key"]] = answer

            # After universal questions, determine type and add specific questions
            if (self.interview_phase == "universal" and
                self.current_question_index == 0):
                # First answer is website type — resolve it
                detected_type = self._resolve_website_type(answer)
                self.session_data["resolved_type"] = detected_type

        self.current_question_index += 1

        # Check if we finished universal questions
        if (self.interview_phase == "universal" and 
            self.current_question_index >= len(self.question_queue)):
            
            # Add type-specific questions
            resolved = self.session_data.get("resolved_type", "other")
            type_questions = QUESTION_TREE.get(resolved, QUESTION_TREE["other"])
            self.question_queue.extend(type_questions)
            self.interview_phase = "type_specific"

        # Check if we finished type-specific questions
        if (self.interview_phase == "type_specific" and
            self.current_question_index >= len(self.question_queue)):
            
            # Add final questions
            self.question_queue.extend(FINAL_QUESTIONS)
            self.interview_phase = "final"

        # Check if all questions are done
        if self.current_question_index >= len(self.question_queue):
            self.interview_phase = "complete"
            return {
                "status": "interview_complete",
                "question": None,
                "question_id": "done",
                "progress": 1.0,
            }

        # Return next question
        next_q = self.question_queue[self.current_question_index]
        total = len(self.question_queue)
        progress = self.current_question_index / total

        return {
            "status": "next_question",
            "question": next_q["question"],
            "question_id": next_q["id"],
            "progress": progress,
        }

    def _resolve_website_type(self, answer: str) -> str:
        """Map user's free-text answer to a WebsiteType."""
        answer_lower = answer.lower()
        
        type_keywords = {
            "portfolio": ["portfolio", "personal site", "my work", "showcase"],
            "landing_page": ["landing", "launch", "product page", "coming soon", "waitlist"],
            "ecommerce": ["store", "shop", "ecommerce", "e-commerce", "sell", "products"],
            "blog": ["blog", "articles", "writing", "posts", "journal"],
            "saas": ["saas", "software", "app", "dashboard", "platform", "tool"],
            "restaurant": ["restaurant", "cafe", "food", "menu", "bakery", "coffee"],
            "agency": ["agency", "studio", "firm", "consultancy", "consulting"],
            "documentation": ["docs", "documentation", "wiki", "knowledge base"],
            "event": ["event", "conference", "wedding", "meetup", "festival"],
            "nonprofit": ["nonprofit", "charity", "ngo", "foundation", "cause"],
            "real_estate": ["real estate", "property", "housing", "apartment", "rental"],
            "education": ["education", "course", "school", "tutorial", "learning"],
        }

        for wtype, keywords in type_keywords.items():
            if any(kw in answer_lower for kw in keywords):
                return wtype

        return "other"

    def select_builder(self) -> BuilderPlatform:
        """Select the best builder based on project requirements."""
        # Check if user specified a preference
        user_pref = self.session_data.get("builder", "").lower()
        
        if "bolt" in user_pref:
            self.selected_builder = BuilderPlatform.BOLT
        elif "lovable" in user_pref:
            self.selected_builder = BuilderPlatform.LOVABLE
        elif "v0" in user_pref:
            self.selected_builder = BuilderPlatform.V0
        elif "replit" in user_pref:
            self.selected_builder = BuilderPlatform.REPLIT
        else:
            # Auto-select based on website type
            resolved_type = self.session_data.get("resolved_type", "other")
            self.selected_builder = BUILDER_RECOMMENDATIONS.get(
                resolved_type, BuilderPlatform.BOLT
            )

        return self.selected_builder

    async def generate_builder_prompt(self) -> str:
        """
        Use Gemini to craft an optimized prompt for the selected builder.

        a production-grade builder prompt.
        """
        builder_config = BUILDER_CONFIGS[self.selected_builder]

# Safety settings

class BrowserAutomation:
    """
    Handles all browser interaction for the web builder agent.
    Opens browser, navigates to builder, pastes prompt, monitors build.
    """

    def __init__(self, sio):
        self.sio = sio
        self.browser_window_title = None
        self.build_monitoring = False
        self._monitor_task = None

    # ─────────────────────────────────────
    # Step 1: Open the Builder
    # ─────────────────────────────────────

    async def open_builder(self, url: str, builder_name: str) -> dict:
        """
        Open the builder URL in the default browser.
        Returns success status and any errors.
        """
        try:
            await self._emit_status("opening_browser", 
                f"Opening {builder_name} in your browser...")

            # Open URL in default browser
            if platform.system() == "Windows":
                subprocess.Popen(
                    ["cmd", "/c", "start", "", url],
                    shell=False
                )
            elif platform.system() == "Darwin":
                subprocess.Popen(["open", url])
            else:
                subprocess.Popen(["xdg-open", url])

            # Wait for browser to open and page to load
            await asyncio.sleep(4)

            await self._emit_status("browser_opened", 
                f"{builder_name} is loading. Waiting for the page to be ready...")

            # Wait for page to fully load
            await self._wait_for_page_load(builder_name)

            return {"success": True, "message": f"{builder_name} opened successfully"}

        except Exception as e:
            logger.error(f"Failed to open browser: {e}")
            return {"success": False, "error": str(e)}

    async def _wait_for_page_load(self, builder_name: str, timeout: int = 20):
        """Wait for the builder page to fully load using screen vision."""
        start = time.time()
        
        while time.time() - start < timeout:
            await asyncio.sleep(2)
            
            # Use screen vision to check if the page has loaded
            screen_description = await analyze_screen(
                question=f"Has the {builder_name} website fully loaded? "
                         f"Can you see an input field or text area where I can type a prompt? "
                         f"Describe what you see on screen."
            )

            if screen_description:
                desc_lower = screen_description.lower()
                # Check for common loaded indicators
                if any(indicator in desc_lower for indicator in 
                       ["textarea", "input", "prompt", "type", "build", 
                        "describe", "what do you want"]):
                    await self._emit_status("page_ready", 
                        f"{builder_name} is ready! I can see the prompt area.")
                    return True

        await self._emit_status("page_timeout", 
            f"{builder_name} seems slow. Proceeding anyway...")
        return False

    # ─────────────────────────────────────
    # Step 2: Find and Click Prompt Area
    # ─────────────────────────────────────

    async def find_and_click_prompt_area(self, builder_config: dict) -> dict:
        """
        Use screen vision to locate the prompt input area and click on it.
        """
        builder_name = builder_config["name"]
        
        await self._emit_status("finding_prompt", 
            f"Looking for the prompt area in {builder_name}...")

        # Use Gemini Vision to find the prompt area
        location = await self._vision_find_element(
            description=f"Find the main text input area or textarea in {builder_name} "
                        f"where I should type my website description prompt. "
                        f"It's usually a large text area in the center of the page "
                        f"with placeholder text like 'Describe what you want to build' "
                        f"or 'What do you want to create'. "
                        f"Return the approximate x,y coordinates of the CENTER "
                        f"of this input area.",
            max_attempts=3
        )

        if not location:
            # Fallback: Try keyboard shortcut or tab navigation
            return await self._fallback_find_prompt(builder_config)

        x, y = location
        
        # Click on the prompt area
        await self._safe_click(x, y)
        await asyncio.sleep(0.5)

        # Verify we clicked the right place
        # (cursor should now be in a text field)
        await self._emit_status("prompt_found", 
            "Found and clicked the prompt area!")

        return {"success": True, "coordinates": [x, y]}

    async def _vision_find_element(
        self, 
        description: str, 
        max_attempts: int = 3
    ) -> Optional[tuple]:
        """
        Use screen vision to find an element and return its coordinates.
        """
        for attempt in range(max_attempts):
            try:
                screenshot = await capture_screenshot()
                
                result = await analyze_screen(
                    question=f"{description}\n\n"
                             f"IMPORTANT: Respond with ONLY two numbers separated "
                             f"by a comma — the x and y pixel coordinates. "
                             f"Example: 960,540\n"
                             f"If you cannot find the element, respond with: NOT_FOUND",
                    screenshot=screenshot
                )

                if result and "NOT_FOUND" not in result.upper():
                    # Parse coordinates
                    parts = result.strip().replace(" ", "").split(",")
                    if len(parts) >= 2:
                        x = int(float(parts[0]))
                        y = int(float(parts[1]))
                        
                        # Sanity check coordinates are on screen
                        screen_w, screen_h = pyautogui.size()
                        if 0 < x < screen_w and 0 < y < screen_h:
                            return (x, y)

                logger.warning(
                    f"Vision attempt {attempt + 1}: couldn't parse coordinates "
                    f"from: {result}"
                )
                
            except Exception as e:
                logger.error(f"Vision find attempt {attempt + 1} failed: {e}")

            await asyncio.sleep(1)

        return None

    async def _fallback_find_prompt(self, builder_config: dict) -> dict:
        """Fallback method: click center of screen and try Tab key."""
        await self._emit_status("fallback_search", 
            "Using fallback method to find prompt area...")

        screen_w, screen_h = pyautogui.size()

        # Strategy 1: Click center of page (most builders put prompt there)
        center_x = screen_w // 2
        center_y = screen_h // 2
        await self._safe_click(center_x, center_y)
        await asyncio.sleep(0.5)

        # Strategy 2: Use Ctrl+L to focus address bar, then Tab to content
        # Actually, just try clicking slightly below center where prompts usually are
        prompt_y = int(screen_h * 0.6)  # 60% down the page
        await self._safe_click(center_x, prompt_y)
        await asyncio.sleep(0.5)

        return {"success": True, "method": "fallback", 
                "coordinates": [center_x, prompt_y]}

    # ─────────────────────────────────────
    # Step 3: Type/Paste the Prompt
    # ─────────────────────────────────────

    async def paste_prompt(self, prompt: str) -> dict:
        """
        Paste the generated prompt into the active text area.
        Uses clipboard paste for speed and reliability.
        """
        await self._emit_status("pasting_prompt", 
            "Pasting the optimized prompt...")

        try:
            import pyperclip
        except ImportError:
            # Fallback without pyperclip
            pyperclip = None

        try:
            if pyperclip:
                # Save current clipboard
                try:
                    old_clipboard = pyperclip.paste()
                except Exception:
                    old_clipboard = ""

                # Copy prompt to clipboard
                pyperclip.copy(prompt)
                await asyncio.sleep(0.2)

                # Paste with Ctrl+V
                pyautogui.hotkey('ctrl', 'v')
                await asyncio.sleep(0.5)

                # Restore old clipboard
                try:
                    pyperclip.copy(old_clipboard)
                except Exception:
                    pass
            else:
                # Fallback: Use pyautogui write (slower, ASCII only)
                # First, try Windows clipboard via subprocess
                if platform.system() == "Windows":
                    process = subprocess.Popen(
                        ['clip'], stdin=subprocess.PIPE
                    )
                    process.communicate(prompt.encode('utf-8'))
                    pyautogui.hotkey('ctrl', 'v')
                    await asyncio.sleep(0.5)
                else:
                    # Last resort: type it character by character
                    pyautogui.write(prompt, interval=0.01)

            prompt_preview = prompt[:100] + "..." if len(prompt) > 100 else prompt
            await self._emit_status("prompt_pasted", 
                f"Prompt pasted! ({len(prompt)} characters)")

            return {"success": True, "length": len(prompt)}

        except Exception as e:
            logger.error(f"Failed to paste prompt: {e}")
            return {"success": False, "error": str(e)}

    # ─────────────────────────────────────
    # Step 4: Submit / Click Run
    # ─────────────────────────────────────

    async def submit_prompt(self, builder_config: dict) -> dict:
        """Submit the prompt by pressing Enter or clicking the submit button."""
        run_method = builder_config.get("run_method", "enter_key")
        builder_name = builder_config["name"]

        await self._emit_status("submitting", 
            f"Submitting prompt to {builder_name}...")

        await asyncio.sleep(0.5)

        if run_method == "enter_key":
            # Most builders use Enter or Ctrl+Enter to submit
            press_key("enter")
            await asyncio.sleep(1)

            # Verify submission happened by checking screen changed
            screen_check = await analyze_screen(
                question=f"Did {builder_name} start processing? "
                         f"Can you see any loading indicators, "
                         f"progress bars, or building status? "
                         f"Or is the prompt still sitting in the input area?"
            )

            if screen_check:
                check_lower = screen_check.lower()
                still_waiting = any(w in check_lower for w in 
                    ["still in input", "prompt still", "hasn't started", "not submitted"])
                
                if still_waiting:
                    # Try Ctrl+Enter instead
                    logger.info("Enter didn't work, trying Ctrl+Enter")
                    pyautogui.hotkey('ctrl', 'enter')
                    await asyncio.sleep(1)

        elif run_method == "click_button":
            # Find and click the Run/Submit/Generate button
            button_location = await self._vision_find_element(
                description=f"Find the submit, run, generate, or send button "
                           f"in {builder_name}. It's usually near the prompt area. "
                           f"Return its center coordinates."
            )

            if button_location:
                await self._safe_click(*button_location)
            else:
                # Fallback to Enter
                press_key("enter")

        await asyncio.sleep(2)
        
        await self._emit_status("submitted", 
            f"Prompt submitted! {builder_name} is now building your website...")

        return {"success": True}

    # ─────────────────────────────────────
    # Step 5: Monitor Build Progress
    # ─────────────────────────────────────

    async def monitor_build(self, builder_config: dict) -> dict:
        """
        Monitor the build process using periodic screen vision analysis.
        Emits progress updates to frontend.
        """
        build_config = builder_config["wait_for_build"]
        builder_name = builder_config["name"]
        check_interval = build_config["check_interval"]
        timeout = build_config["timeout"]
        completion_indicators = build_config["completion_indicators"]
        building_indicators = build_config["building_indicators"]

        self.build_monitoring = True
        start_time = time.time()
        last_status = ""
        check_count = 0

        await self._emit_status("monitoring", 
            f"Monitoring {builder_name} build progress...")

        while self.build_monitoring:
            elapsed = time.time() - start_time

            if elapsed > timeout:
                await self._emit_status("build_timeout", 
                    f"Build monitoring timed out after {timeout}s. "
                    f"The builder might still be working — check your browser.")
                return {
                    "success": False, 
                    "reason": "timeout",
                    "elapsed": elapsed
                }

            check_count += 1

            try:
                screen_description = await analyze_screen(
                    question=f"I'm building a website on {builder_name}. "
                             f"Look at the screen and tell me:\n"
                             f"1. Is the website still being built/generated?\n"
                             f"2. Is there a preview visible?\n"
                             f"3. Is it complete/finished/deployed?\n"
                             f"4. Are there any errors?\n"
                             f"5. What percentage complete does it look?\n"
                             f"Be concise."
                )

                if screen_description:
                    desc_lower = screen_description.lower()

                    # Check for completion
                    is_complete = any(
                        indicator in desc_lower 
                        for indicator in completion_indicators
                    )

                    # Check for errors
                    has_error = any(
                        err in desc_lower 
                        for err in ["error", "failed", "crashed", "limit"]
                    )

                    # Check if still building
                    is_building = any(
                        indicator in desc_lower 
                        for indicator in building_indicators
                    )

                    # Estimate progress
                    progress = min(0.95, elapsed / timeout)
                    if is_complete:
                        progress = 1.0

                    # Emit progress update
                    status_msg = screen_description[:200]
                    if status_msg != last_status:
                        await self._emit_progress(
                            progress=progress,
                            message=status_msg,
                            phase="building" if is_building else 
                                  "complete" if is_complete else
                                  "error" if has_error else "checking"
                        )
                        last_status = status_msg

                    if is_complete:
                        await self._emit_status("build_complete",
                            f"Your website is ready! {builder_name} has "
                            f"finished building. Check your browser for "
                            f"the preview!")
                        return {
                            "success": True,
                            "elapsed": elapsed,
                            "checks": check_count,
                            "final_status": screen_description
                        }

                    if has_error:
                        await self._emit_status("build_error",
                            f"There seems to be an error: {status_msg}")
                        # Don't return — errors might be recoverable
                        # Let it keep monitoring

            except Exception as e:
                logger.error(f"Monitor check failed: {e}")

            await asyncio.sleep(check_interval)

        return {"success": False, "reason": "cancelled"}

    def stop_monitoring(self):
        """Stop the build monitoring loop."""
        self.build_monitoring = False

    # ─────────────────────────────────────
    # Utility Methods
    # ─────────────────────────────────────

    async def _safe_click(self, x: int, y: int):
        """Click with safety checks."""
        screen_w, screen_h = pyautogui.size()
        x = max(1, min(x, screen_w - 1))
        y = max(1, min(y, screen_h - 1))
        
        pyautogui.click(x, y)
        logger.info(f"Clicked at ({x}, {y})")

    async def _emit_status(self, phase: str, message: str):
        """Emit build status to frontend."""
        await self.sio.emit("web_builder_status", {
            "phase": phase,
            "message": message,
            "timestamp": time.time(),
        })

    async def _emit_progress(self, progress: float, message: str, phase: str):
        """Emit build progress to frontend."""
        await self.sio.emit("web_builder_progress", {
            "progress": progress,
            "message": message,

"""
Web Builder Orchestrator — Connects the agent, prompt generator,
and browser automation into a seamless flow.
"""

import asyncio
from typing import Optional

from web_builder_agent import WebBuilderAgent, BUILDER_CONFIGS
from browser_automation import BrowserAutomation
from logger import get_logger

logger = get_logger("web_builder_orchestrator")


class WebBuilderOrchestrator:
    """
    Full orchestration of the website building flow.
    Called from server.py tool dispatch or directly from AudioLoop.
    """

    def __init__(self, sio, gemini_client=None):
        self.sio = sio
        self.agent = WebBuilderAgent(sio, gemini_client)
        self.browser = BrowserAutomation(sio)
        self.state = "idle"
        self._build_task: Optional[asyncio.Task] = None

    # ─────────────────────────────────────
    # Tool Functions (called by Gemini)
    # ─────────────────────────────────────

    async def start_website_project(self) -> dict:
        """
        Tool: Start a new website building project.
        Begins the interview phase.
        """
        self.state = "interviewing"
        first_question = self.agent.start_interview()

        await self.sio.emit("web_builder_status", {
            "phase": "interview_started",
            "message": "Starting website project interview...",
        })

        return {
            "status": "interview_started",
            "instruction_to_soda": (
                "I've started the website builder interview. "
                "Ask the user the following question and wait for their answer. "
                "After they answer, call web_builder_answer with their response.\n\n"
                f"QUESTION: {first_question}"
            ),
        }

    async def process_answer(self, answer: str) -> dict:
        """
        Tool: Process user's answer during interview.
        Returns next question or triggers build.
        """
        result = self.agent.process_answer(answer)

        if result["status"] == "interview_complete":
            # Interview done — generate prompt and start building
            return await self._prepare_and_build()

        return {
            "status": "next_question",
            "progress": result["progress"],
            "instruction_to_soda": (
                f"The user answered. Here's the next question to ask them:\n\n"
                f"QUESTION: {result['question']}\n\n"
                f"Progress: {result['progress']:.0%} complete."
            ),
        }

    async def _prepare_and_build(self) -> dict:
        """Generate prompt, select builder, and start automation."""
        self.state = "preparing"

        # Select builder
        builder = self.agent.select_builder()
        builder_config = BUILDER_CONFIGS[builder]
        builder_name = builder_config["name"]

        await self.sio.emit("web_builder_status", {
            "phase": "generating_prompt",
            "message": f"Selected {builder_name}. Crafting the perfect prompt...",
            "builder": builder_name,
        })

        # Generate optimized prompt
        prompt = await self.agent.generate_builder_prompt()

        await self.sio.emit("web_builder_status", {
            "phase": "prompt_ready",
            "message": f"Prompt ready! ({len(prompt)} chars). Opening {builder_name}...",
            "prompt_preview": prompt[:300],
        })

        # Start the automation flow in background
        self._build_task = asyncio.create_task(
            self._run_build_automation(builder_config, prompt)
        )

        return {
            "status": "building",
            "builder": builder_name,
            "prompt_length": len(prompt),
            "instruction_to_soda": (
                f"I'm now automating {builder_name} to build the website. "
                f"I'll open the browser, paste the prompt, and start the build. "
                f"Tell the user to sit back and watch the magic happen! "
                f"I'll keep them updated on progress.\n\n"
                f"Builder: {builder_name}\n"
                f"Prompt length: {len(prompt)} characters\n"
                f"Estimated build time: 1-3 minutes"
            ),
        }

    async def _run_build_automation(self, builder_config: dict, prompt: str):
        """
        The full automation sequence. Runs as a background task.
        """
        builder_name = builder_config["name"]
        
        try:
            # Step 1: Open browser
            result = await self.browser.open_builder(
                builder_config["url"], builder_name
            )
            if not result["success"]:
                await self._emit_failure(
                    f"Failed to open browser: {result.get('error')}"
                )
                return

            # Step 2: Find and click prompt area
            result = await self.browser.find_and_click_prompt_area(builder_config)
            if not result["success"]:
                await self._emit_failure("Couldn't find the prompt input area")
                return

            # Brief pause to ensure focus
            await asyncio.sleep(0.5)

            # Step 3: Paste the prompt
            result = await self.browser.paste_prompt(prompt)
            if not result["success"]:
                await self._emit_failure(
                    f"Failed to paste prompt: {result.get('error')}"
                )
                return

            # Step 4: Submit
            result = await self.browser.submit_prompt(builder_config)

            # Step 5: Monitor build
            self.state = "monitoring"
            result = await self.browser.monitor_build(builder_config)

            if result["success"]:
                self.state = "complete"
                await self.sio.emit("web_builder_status", {
                    "phase": "complete",
                    "message": (
                        f"Website build complete! "
                        f"Took {result['elapsed']:.0f} seconds. "
                        f"Check your browser for the preview!"
                    ),
                    "elapsed": result["elapsed"],
                })
            else:
                self.state = "failed"
                await self._emit_failure(
                    f"Build didn't complete: {result.get('reason')}"
                )

        except asyncio.CancelledError:
            self.state = "cancelled"
            await self.sio.emit("web_builder_status", {
                "phase": "cancelled",
                "message": "Website building was cancelled.",
            })
        except Exception as e:
            logger.error(f"Build automation failed: {e}")
            self.state = "error"
            await self._emit_failure(str(e))

    async def _emit_failure(self, message: str):
        """Emit failure status."""
        await self.sio.emit("web_builder_status", {
            "phase": "error",
            "message": f"Something went wrong: {message}. "
                       f"You might need to do it manually.",
        })

    def cancel_build(self):
        """Cancel the current build process."""
        self.browser.stop_monitoring()
        if self._build_task:
            self._build_task.cancel()
        self.state = "cancelled"

    def get_status(self) -> dict:
        """Get current orchestrator status."""
        return {
            "state": self.state,
            "session": self.agent.get_session_summary(),
        }

class AutomationSafety:
    """Prevent automation from going rogue."""

    MAX_CLICKS_PER_SESSION = 20
    MAX_KEYSTROKES_PER_SESSION = 5000
    FORBIDDEN_URLS = [
        "chrome://", "about:", "file://",
        "bank", "paypal", "stripe.com/dashboard"
    ]

    def __init__(self):
        self.click_count = 0
        self.keystroke_count = 0

    def check_click(self) -> bool:
        self.click_count += 1
        if self.click_count > self.MAX_CLICKS_PER_SESSION:
            raise SafetyError(
                f"Too many clicks ({self.click_count}). Aborting."
            )
        return True

    def check_url(self, url: str) -> bool:
        for forbidden in self.FORBIDDEN_URLS:
            if forbidden in url.lower():
                raise SafetyError(
                    f"Forbidden URL pattern: {forbidden}"
                )
        return True

    def check_keystroke(self, text: str) -> bool:
        self.keystroke_count += len(text)
        if self.keystroke_count > self.MAX_KEYSTROKES_PER_SESSION:
            raise SafetyError(
                f"Too many keystrokes ({self.keystroke_count}). Aborting."
            )
        return True


class SafetyError(Exception):
    pass
