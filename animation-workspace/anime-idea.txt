1. ⚡ Startup Briefing — "COLD BOOT SEQUENCE"
Concept: Nuclear Submarine Cold Start + Fighter Jet Avionics Boot
Imagine a nuclear submarine's reactor coming online — systems cascading to life one by one, each subsystem reporting status through a command-line-style HUD that transforms into a polished holographic display.

  as they travel (like a line being drawn in real-time)

2. Panel Transition System
text

ENTERING:
  - Panels materialize with a "holographic build" effect:
    1. Wireframe outline appears (50ms)
    2. Fill materializes inside (150ms) — like static clearing to image
    3. Content fades in (200ms)
    Total: ~400ms per panel

EXITING:
  - Reverse of entering:
    1. Content fades (100ms)
    2. Fill dissolves to static/noise (150ms)
    3. Wireframe collapses to center point (100ms)
    Total: ~350ms per panel

REARRANGING:
  - Panels slide smoothly to new positions
  - Spring physics: slight overshoot + settle
  - cubic-bezier(0.34, 1.56, 0.64, 1)

// particles.js — Ambient particle system

class HUDParticleSystem {
  constructor(canvas, options = {}) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d');
    this.particles = [];
    this.options = {
      count: options.count || 30,
      color: options.color || '#00F0FF',
      speed: options.speed || 0.5,
      size: options.size || 2,
      glow: options.glow || true,
      type: options.type || 'float' // 'float', 'rise', 'spark', 'rain'
    };
  }

  createParticle() {
    return {
      x: Math.random() * this.canvas.width,
      y: Math.random() * this.canvas.height,
      vx: (Math.random() - 0.5) * this.options.speed,
      vy: (Math.random() - 0.5) * this.options.speed,
      size: Math.random() * this.options.size + 0.5,
      opacity: Math.random() * 0.5 + 0.2,
      life: Math.random() * 200 + 100
    };
  }

  update() {
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
    
    this.particles.forEach((p, i) => {
      p.x += p.vx;
      p.y += p.vy;
      p.life--;
      p.opacity = Math.max(0, p.opacity - 0.002);

      if (p.life <= 0) {
        this.particles[i] = this.createParticle();
      }

      // Draw
      this.ctx.beginPath();
      this.ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
      this.ctx.fillStyle = `${this.options.color}${Math.floor(p.opacity * 255).toString(16).padStart(2, '0')}`;
      
      if (this.options.glow) {
        this.ctx.shadowBlur = p.size * 4;
        this.ctx.shadowColor = this.options.color;
      }
      
      this.ctx.fill();
    });

    requestAnimationFrame(() => this.update());
  }

  start() {
    for (let i = 0; i < this.options.count; i++) {
      this.particles.push(this.createParticle());
    }
    this.update();
  }

  // Change particle behavior for different modes
  setMode(mode) {
    switch(mode) {
      case 'break':
        this.options.color = '#F5C842';
        this.options.speed = 0.2;
        this.options.count = 10;
        break;
      case 'combat':
        this.options.color = '#00F0FF';
        this.options.speed = 1;
        this.options.count = 50;
        break;
      case 'night':
        this.options.color = '#FFFFFF';
        this.options.speed = 0.1;
        this.options.count = 40;
        this.options.size = 1;
        break;
    }
  }
}
